function totalDays(){
    var dropOff = new Date(document.getElementById("date2").value); 
    var pickUp = new Date(document.getElementById("date").value);
    return parseInt((dropOff - pickUp) / (24 * 3600 * 1000));
}

function calNumDays(){
        if(document.getElementById("date2")){
            document.getElementById("numOfDays").value=totalDays();
        } 
}

function totalRentalCost(){
    var numOfDays = document.getElementById("numOfDay").value;
    var carModelForRent = document.getElementById("model").value;
    var carPriceForRent = document.getElementById("carPrice").value;
    document.getElementById("flatPrice").innerHTML = "<h2> $" + (numOfDays * carPriceForRent).toFixed(2) + "</h2>";
    document.getElementById("daysRenting").innerHTML = "<h2>" + numOfDays + "</h2>";
    document.getElementById("tax").innerHTML = "<h2> $" + (0.05 * (numOfDays * carPriceForRent)).toFixed(2) + "</h2>";
    document.getElementById("total").innerHTML = "<h2> $" + (1.05 * (numOfDays * carPriceForRent)).toFixed(2) + "</h2>";
    
}
